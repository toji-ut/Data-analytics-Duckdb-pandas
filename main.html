          region = 'southeast'
    GROUP BY region
''').df()

palette_color = \
    sns.color_palette('pastel')
plt.figure (figsize = (5, 5))

df = conn.execute ('''
    SELECT count (*) as Count,
           smoker
    FROM insurance
    GROUP BY smoker
    ORDER BY Count DESC
''').df()

plt.pie('Count',
        labels = 'smoker',
        colors = palette_color,
        data = df,
        autopct = '%.0f%%',)

plt.legend(df['smoker'], loc="best")

def fmt(x):
    # display percentage followed by number
    return '{:.2f}%\n({:.0f})'.format(
        x, total * x / 100)
